4. 핵심 코드

4.1 image_sink · condvar pub/sub

class ImageSink:
    def __init__(self):
        self._cond = threading.Condition()
        self._latest: dict[str, dict] = {}

    def push(self, camera_id, encoding, w, h, data, sequence, captured_at_iso):
        with self._cond:
            self._latest[camera_id] = {...}
            self._cond.notify_all()            # 구독자 전원 깨우기

    def wait_new(self, camera_id, after_sequence, timeout=10.0):
        with self._cond:
            cur = self._latest.get(camera_id)
            if cur and int(cur["sequence"]) > after_sequence:
                return cur
            self._cond.wait(timeout=timeout)
            cur = self._latest.get(camera_id)
            if cur and int(cur["sequence"]) > after_sequence:
                return cur
            return None

4.2 서버 WatchCameraFrames

def WatchCameraFrames(self, request, context):
    cam = request.camera_id or ""
    if not cam:
        context.set_code(grpc.StatusCode.INVALID_ARGUMENT); return
    last_seq = int(request.after_sequence)
    while context.is_active():
        frame = image_sink.wait_new(cam, last_seq, timeout=10.0)
        if frame is None:
            continue    # keepalive timeout — 클라이언트 여전히 살아있으면 계속
        last_seq = int(frame.get("sequence", 0))
        yield self._frame_to_response(cam, frame)

4.3 PyQt CameraFrameWorker

class CameraFrameWorker(QObject):
    frame_received = pyqtSignal(bytes, str, int, str)

    def run(self):
        while not self._stop:
            try:
                client = ManagementClient()
                stream = client._stub.WatchCameraFrames(
                    management_pb2.WatchFramesRequest(
                        camera_id=self._camera_id,
                        after_sequence=self._last_sequence,
                    ))
                for resp in stream:
                    if self._stop: break
                    if not resp.available: continue
                    self._last_sequence = int(resp.sequence)
                    self.frame_received.emit(
                        bytes(resp.data), resp.encoding or "jpeg",
                        self._last_sequence,
                        resp.received_at.iso8601,
                    )
            except Exception as exc:
                if self._stop: break
                time.sleep(self._reconnect)   # 3s backoff

6. 알려진 한계 · 향후

알려진 한계:
  • PyQt 앱 종료 시 worker 가 조용히 쌓일 수 있음 → QualityPage.shutdown() 추가해뒀지만 main_window 가 호출 안 함 (daemon thread 라 프로세스 종료 시 정리됨)
  • 카메라 ID 는 현재 하드코딩(환경변수) 1개 — 다중 카메라는 worker 풀 필요
  • HTTP/2 흐름 제어로 클라이언트가 느리면 서버 yield 가 blocking. 현재 2fps 85KB 기준 문제 없음
  • mTLS 활성 시 관리 서버 재기동 필요 (동일한 서버 재기동 절차)
