① 개요

Execution Monitor / Exception Handler는 전체 실행 루프를 닫는 피드백 컴포넌트입니다. Robot/Cell Executors로부터 결과를 수신하여 시스템 상태를 갱신하고, 실패 시 적절한 복구 전략을 선택합니다.

1.1 역할

  • Executor 실행 결과 수신 (성공/실패)
  • task 상태 전이 (running → completed/failed)
  • 실패 원인 분류 및 복구 전략 결정
  • 재시도 / 재할당 / 사용자 개입 요청
  • Lane Queue에 work_order 완료 통지
  • WebSocket으로 UI에 실시간 알림

② 수신 이벤트

Execution Monitor는 여러 소스로부터 이벤트를 구독합니다:

이벤트소스처리
task.completed Executor 성공 시 Workflow에 알림, 실패 시 복구 판단
task.timeout Monitor 자체 timer 장시간 응답 없음 → 강제 실패 처리
robot.error Executor 하드웨어 에러 해당 robot unavailable 마킹, task 재할당
resource.stale Resource Reservation Manager stale 예약 경보
quality.fail Vision Executor 불량 판정 시 sorting task 트리거

③ 성공 처리

def on_task_completed(result: ExecutionResult):
    task = db.query(Task).get(result.task_id)

    # 1. Task 상태 전이
    task.state = TaskState.COMPLETED
    task.ended_at = datetime.utcnow()
    task.actual_duration_sec = result.elapsed_sec
    task.output = result.output

    # 2. Executor robot 해제
    robot = fleet[task.assigned_to]
    robot.state = "idle"
    robot.current_task_id = None

    # 3. Resource 해제
    if task.requires_station:
        resource_mgr.release(task.station_id, task.assigned_to)

    # 4. 공정 이력 기록
    db.add(ProductionProcessLog(
        work_order_id=task.work_order_id,
        process_type=task.type.value,
        process_status="completed",
        started_at=task.started_at,
        ended_at=task.ended_at,
        result_note=json.dumps(result.output),
    ))
    db.commit()

    # 5. Workflow에 알림 → dependency 해결
    workflow_mgr.on_task_completed(task.task_id, success=True)

    # 6. WebSocket 브로드캐스트
    broadcast_task_update(task)

④ 실패 분류와 복구 전략

실패는 원인별로 다르게 처리:

실패 타입판단 기준복구 전략
일시적 (Transient) 센서 노이즈, 네트워크 지연 동일 robot에 재시도 (최대 3회)
Robot 문제 (Robot-bound) 배터리 방전, 센서 고장 robot unavailable, 다른 robot에 재할당
환경 문제 (Environmental) 장애물, 경로 막힘 10초 후 재시도, 반복 시 관리자 개입 요청
영구 실패 (Permanent) 품질 불량, 데이터 불일치 task FAILED, work_order FAILED, 관리자 알람
상위 시스템 에러 예약 충돌, DB 에러 task를 Ready Pool로 반환, 원인 로그

4.1 분류 로직

def classify_failure(result: ExecutionResult) -> FailureClass:
    reason = result.failure_reason.lower()

    if "timeout" in reason or "network" in reason:
        return FailureClass.TRANSIENT
    if "battery" in reason or "hardware" in reason:
        return FailureClass.ROBOT_BOUND
    if "obstacle" in reason or "path not found" in reason:
        return FailureClass.ENVIRONMENTAL
    if "quality" in reason or "inspection" in reason:
        return FailureClass.PERMANENT
    return FailureClass.UNKNOWN

⑤ 재시도 정책

class RetryPolicy:
    MAX_RETRIES = 3
    BACKOFF_BASE_SEC = 2

    def should_retry(self, task: Task, failure: FailureClass) -> bool:
        if task.retry_count >= self.MAX_RETRIES:
            return False
        if failure == FailureClass.PERMANENT:
            return False
        return True

    def next_delay(self, retry_count: int) -> float:
        # Exponential backoff: 2s, 4s, 8s
        return self.BACKOFF_BASE_SEC * (2 ** retry_count)

def handle_failure(result: ExecutionResult):
    task = db.query(Task).get(result.task_id)
    failure_class = classify_failure(result)

    if retry_policy.should_retry(task, failure_class):
        delay = retry_policy.next_delay(task.retry_count)
        task.retry_count += 1
        task.state = TaskState.READY
        db.commit()

        # 일정 시간 후 Pool에 재삽입
        scheduler.enter(delay, lambda: ready_pool.insert(task))

        log.info(f"Retrying {task.task_id} in {delay}s "
                 f"(attempt {task.retry_count}/{retry_policy.MAX_RETRIES})")
    else:
        # 영구 실패
        task.state = TaskState.FAILED
        task.failure_reason = result.failure_reason
        db.commit()

        # Workflow에 실패 전파 → work_order FAILED
        workflow_mgr.on_task_completed(task.task_id, success=False)

        # 관리자 알람
        emit_alert(
            level="critical",
            message=f"Task {task.task_id} permanently failed",
            details=result.failure_reason,
        )

    payload = {
        "event": "task_update",
        "task_id": task.task_id,
        "work_order_id": task.work_order_id,

6.3 알림 심각도

  • Info: 정기 점검, 정상 완료
  • Warning: 배터리 30% 이하, 일시 실패 후 재시도
  • Critical: Permanent failure, robot unavailable, stale reservation
  • Emergency: 배터리 15% 이하, 설비 고장

⑦ 구현 스케치

class ExecutionMonitor:
    def __init__(self, db, workflow_mgr, ready_pool,
                 resource_mgr, fleet, ws_manager):
        self.db = db
        self.workflow_mgr = workflow_mgr
        self.ready_pool = ready_pool
        self.resource_mgr = resource_mgr
        self.fleet = fleet
        self.ws = ws_manager
        self.retry_policy = RetryPolicy()

    def on_execution_result(self, result: ExecutionResult):
        task = self.db.query(Task).get(result.task_id)
        if not task:
            log.warning(f"Unknown task {result.task_id}")
            return

        # 공정 로그 기록
        self._record_process_log(task, result)

        if result.success:
            self._handle_success(task, result)
        else:
            self._handle_failure(task, result)

    def _handle_success(self, task, result):
        task.state = TaskState.COMPLETED
        task.ended_at = datetime.utcnow()
        task.output = result.output

        # 자원 해제
        robot = self.fleet[task.assigned_to]
        robot.state = "idle"
        robot.current_task_id = None
        if task.station_id:
            self.resource_mgr.release(task.station_id, task.assigned_to)

        self.db.commit()

        # Workflow에 dependency 해결 트리거
        self.workflow_mgr.on_task_completed(task.task_id, success=True)

        # UI 갱신
        self._broadcast(task)

    def _handle_failure(self, task, result):
        failure_class = classify_failure(result)
        log.warning(
            f"Task {task.task_id} failed: {result.failure_reason} "
            f"(class={failure_class.value})"
        )

        if failure_class == FailureClass.ROBOT_BOUND:
            # 로봇 문제 → unavailable 처리
            robot = self.fleet[task.assigned_to]
            robot.state = "unavailable"
            emit_alert("critical",
                       f"Robot {robot.id} marked unavailable")

        if self.retry_policy.should_retry(task, failure_class):
            self._retry(task, failure_class)
        else:
            self._mark_permanent_failure(task, result)

    def _retry(self, task, failure_class):
        delay = self.retry_policy.next_delay(task.retry_count)
        task.retry_count += 1
        task.state = TaskState.READY
        task.assigned_to = None  # 재할당 가능
        self.db.commit()

        # scheduler에 delayed insert 예약
        threading.Timer(delay, lambda: self.ready_pool.insert(task)).start()
        log.info(f"Retry scheduled for {task.task_id} in {delay}s")

    def _mark_permanent_failure(self, task, result):
        task.state = TaskState.FAILED
        task.failure_reason = result.failure_reason
        task.ended_at = datetime.utcnow()
        self.db.commit()

        self.workflow_mgr.on_task_completed(task.task_id, success=False)
        emit_alert("critical",
                   f"Task {task.task_id} permanently failed: "
                   f"{result.failure_reason}")

    def _broadcast(self, task):
        self.ws.broadcast("/ws/dashboard", {
            "event": "task_update",
            "task_id": task.task_id,
            "state": task.state.value,
        })
